import java.util.Scanner;
public class arithmetic{
  public static void main(String args[]){

    int a=sc.nextInt();
    int b=sc.nextInt();
    int c=a+b;
    int d=a-b;
    int e=a*b;
    System.out.println("addition "+c);
    System.out.println("subtraction "+d);
    System.out.println("multiplication "+e);
  }
}
